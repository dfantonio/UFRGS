365 void renderizaEstoque(Jogo *jogo) {
366  int descarte = jogo->pos_estoque - 1;
367  int ncartas = 0;
368  int ultimaCarta = 0;
369  for (int aux = 0; aux < 24; aux++) { // Conta quantas cartas existem no estoque/descarte
370  if (jogo->estoque[aux].numero != 0) ncartas++;
371  }
372  for (int aux = 0; aux < 24; aux++) { // Encontra a posição da última carta
373  if (jogo->estoque[aux].numero != 0) ultimaCarta = aux + 1;
374  }
375 
376  while (jogo->estoque[descarte].numero == 0 && jogo->pos_estoque > 0 && jogo->pos_estoque <= 24) { // cartas removidas tem valor 0, e são puladas. menos a posição inicial que também é 0
377  jogo->pos_estoque++;
378  descarte++;
379  }
380 
381  if (jogo->pos_estoque > 24) { // quando chega na posição 25 ou maior ele volta para o começo
382  jogo->pos_estoque = 0;
383  descarte = -1;
384  somaScore(jogo, PONT_FLIP_ESTOQUE);
385  }
386 
387  if (descarte >= 0) jogo->estoque[descarte].visivel = true; // Toda carta no descarte deve estar visível
388  jogo->estoque[jogo->pos_estoque].visivel = false;
389 
390  if (jogo->pos_estoque != ultimaCarta && ncartas != 0) renderizaCarta(&jogo->estoque[jogo->pos_estoque], 4, 6); // renderiza o estoque atualizado
391  if (descarte >= 0) renderizaCarta(&jogo->estoque[descarte], 14, 6); // renderiza o descarte atualizado
392 }
